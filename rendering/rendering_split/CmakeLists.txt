#thnx AI

cmake_minimum_required(VERSION 3.14)

project(rendering LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release)
endif()

# -------------------------
# Serial target
# -------------------------
add_executable(serial_renderer
    src/serial/main.cpp
)

target_include_directories(serial_renderer PRIVATE
    ${CMAKE_SOURCE_DIR}/include
)

target_compile_options(serial_renderer PRIVATE
    $<$<CONFIG:Release>:-O2>
    $<$<CONFIG:Debug>:-g>
)

# -------------------------
# Optional CUDA target
# -------------------------
include(CheckLanguage)
check_language(CUDA)

if(CMAKE_CUDA_COMPILER)
    enable_language(CUDA)

    add_executable(cuda_renderer
        src/parallel/main.cu
        src/parallel/rayTrace.cu
    )

    target_include_directories(cuda_renderer PRIVATE
        ${CMAKE_SOURCE_DIR}/include
    )

    set_target_properties(cuda_renderer PROPERTIES
        CUDA_STANDARD 17
        CUDA_STANDARD_REQUIRED ON
        CUDA_SEPARABLE_COMPILATION ON
    )

    target_compile_options(cuda_renderer PRIVATE
        $<$<COMPILE_LANGUAGE:CXX>:
            $<$<CONFIG:Release>:-O2>
            $<$<CONFIG:Debug>:-g>
        >
        $<$<COMPILE_LANGUAGE:CUDA>:
            $<$<CONFIG:Release>:-O2>
            $<$<CONFIG:Debug>:-G>
        >
    )

    message(STATUS "CUDA found: building cuda_renderer")
else()
    message(STATUS "CUDA not found: skipping cuda_renderer")
endif()
